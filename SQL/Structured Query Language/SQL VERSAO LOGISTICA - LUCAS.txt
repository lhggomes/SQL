SELECT RT.CODFILIAL, FL.RAZAOSOCIAL AS EMPRESA, CG.CONTA AS DESPESA,  CC.DESCRICAO AS CENTRO_CUSTO, SUM (RT.VALOR) AS VALOR_CC  
FROM PCRATEIOCENTROCUSTO RT, PCCENTROCUSTO CC, PCFILIAL FL, PCCONTA CG
WHERE RT.DTLANC BETWEEN :DATA1 AND :DATA2 
AND RT.CODCONTA = CG.CODCONTA
AND RT.CODFILIAL = :CODFILIAL 
AND RT.CODIGOCENTROCUSTO = CC.CODIGOCENTROCUSTO 
AND RT.CODFILIAL = FL.CODIGO
AND RT.CODIGOCENTROCUSTO IN(:CENTROCUSTO)
GROUP BY RT.CODFILIAL, FL.RAZAOSOCIAL, CG.CONTA,  RT.CODIGOCENTROCUSTO, CC.DESCRICAO
ORDER BY  5
