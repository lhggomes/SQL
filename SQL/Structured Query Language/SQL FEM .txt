SELECT M.CODFILIAL, NFE.DTENT, M.NUMNOTA, 
M.CODPROD, P.DESCRICAO, M.QTCONT, M.VLITEM, 
M.BASECALCFCP, M.PERCFCP, M.VLFCP 
--REPLACE(NOMEUSUARIOLOGADO,'LUCAS.HENRIQUE', 'LUCASHENRIQUE') AS USUARIO
FROM PCMOVCIAP M, PCNFENT NFE, PCPRODCIAP P
WHERE M.NUMTRANSENT = NFE.NUMTRANSENT
AND M.CODPROD = P.CODPROD
AND NFE.DTENT BETWEEN :DTINICIO AND :DTFIM
AND M.DTMOV BETWEEN :DTINICIO AND :DTFIM
AND M.CODFILIAL = :CODFILIAL
AND NVL(M.VLFCP,0) BETWEEN 0.01 AND 1999999
ORDER BY NFE.DTENT
